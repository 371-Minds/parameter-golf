Forest plot script for corrected comparisons

Yes — this is the right visualization layer now.

A forest plot is ideal because it shows, for each cross-model comparison:

effect estimate
confidence interval
direction
corrected significance

all in one figure.

What this script expects

Input:

cross_model_permutation_corrected.json

Generated by the corrected permutation script.

What it plots

For each experiment, it creates one forest plot where each row is a test:

metric
model_a vs model_b
effect estimate: A - B
95% bootstrap CI
corrected significance markers

By default it uses BH-FDR significance as the highlighted correction, but you can switch to:

bh
holm
bonferroni
raw
Visual semantics
point = estimated difference
horizontal line = 95% CI
vertical line at 0 = no difference
color:
blue = significant under chosen correction
gray = not significant
annotation:
corrected p-value
shared pair count

So you immediately see:

which model comparisons are robust
effect direction
uncertainty width
whether effects survive correction
Python script
import os
import json
import math
import argparse
import matplotlib.pyplot as plt


# ============================================================
# IO
# ============================================================

def load_json(path):
    with open(path, "r", encoding="utf-8") as f:
        return json.load(f)


# ============================================================
# Helpers
# ============================================================

def ensure_dir(path):
    os.makedirs(path, exist_ok=True)

def fmt(x, digits=4):
    if x is None:
        return "n/a"
    return f"{x:.{digits}f}"

def get_corrected_p(row, correction):
    corr = row.get("multiple_comparison_correction", {})
    if correction == "bh":
        return corr.get("bh_fdr_p")
    elif correction == "holm":
        return corr.get("holm_p")
    elif correction == "bonferroni":
        return corr.get("bonferroni_p")
    elif correction == "raw":
        return corr.get("raw_p")
    else:
        raise ValueError(f"Unknown correction: {correction}")

def get_significant(row, correction):
    corr = row.get("multiple_comparison_correction", {})
    if correction == "bh":
        return corr.get("significant_bh_fdr", False)
    elif correction == "holm":
        return corr.get("significant_holm", False)
    elif correction == "bonferroni":
        return corr.get("significant_bonferroni", False)
    elif correction == "raw":
        return corr.get("significant_raw", False)
    else:
        raise ValueError(f"Unknown correction: {correction}")

def comparison_label(row):
    return f"{row['metric']} | {row['model_a']} vs {row['model_b']}"


# ============================================================
# Data extraction
# ============================================================

def extract_rows_for_experiment(data, experiment):
    result = data.get(experiment, {})
    rows = result.get("pairwise_comparisons", [])
    extracted = []

    for row in rows:
        ci = row.get("difference_ci", {})
        extracted.append({
            "label": comparison_label(row),
            "metric": row.get("metric"),
            "model_a": row.get("model_a"),
            "model_b": row.get("model_b"),
            "estimate": row.get("difference_a_minus_b"),
            "ci_low": ci.get("ci_low"),
            "ci_high": ci.get("ci_high"),
            "n_shared_pairs": row.get("n_shared_pairs"),
            "raw_row": row,
        })

    return extracted


# ============================================================
# Sorting
# ============================================================

def sort_rows(rows):
    def key(r):
        est = r["estimate"]
        if est is None:
            est = -999
        return (r["metric"], est)
    return sorted(rows, key=key, reverse=True)


# ============================================================
# Plotting
# ============================================================

def forest_plot(rows, title, correction, out_path, figsize_width=12):
    if not rows:
        return

    rows = sort_rows(rows)
    n = len(rows)
    fig_height = max(4, 0.55 * n + 1.5)

    fig, ax = plt.subplots(figsize=(figsize_width, fig_height))

    y_positions = list(range(n))[::-1]

    # compute x-limits
    xs = []
    for r in rows:
        if r["estimate"] is not None:
            xs.append(r["estimate"])
        if r["ci_low"] is not None:
            xs.append(r["ci_low"])
        if r["ci_high"] is not None:
            xs.append(r["ci_high"])

    if xs:
        xmin = min(xs)
        xmax = max(xs)
        pad = max(0.05, 0.1 * max(abs(xmin), abs(xmax), 0.1))
        xleft = xmin - pad
        xright = xmax + pad
    else:
        xleft, xright = -1, 1

    ax.axvline(0, color="black", linestyle="--", linewidth=1)

    for y, r in zip(y_positions, rows):
        est = r["estimate"]
        lo = r["ci_low"]
        hi = r["ci_high"]
        sig = get_significant(r["raw_row"], correction)
        pval = get_corrected_p(r["raw_row"], correction)

        color = "#2C7FB8" if sig else "#9E9E9E"

        if lo is not None and hi is not None:
            ax.hlines(y, lo, hi, color=color, linewidth=2)

        if est is not None:
            ax.plot(est, y, "o", color=color, markersize=7)

        annot = f"p={fmt(pval, 4)}, n={r['n_shared_pairs']}"
        ax.text(
            xright,
            y,
            annot,
            va="center",
            ha="left",
            fontsize=9,
            color=color,
        )

    ax.set_yticks(y_positions)
    ax.set_yticklabels([r["label"] for r in rows], fontsize=9)
    ax.set_xlim(xleft, xright + 0.28 * (xright - xleft))
    ax.set_xlabel("Effect difference (Model A - Model B)")
    ax.set_title(f"{title}\nSignificance shown using: {correction}")
    plt.tight_layout()
    plt.savefig(out_path, dpi=180, bbox_inches="tight")
    plt.close()


# ============================================================
# Main
# ============================================================

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--input", required=True, help="cross_model_permutation_corrected.json")
    parser.add_argument("--output-dir", default="forest_plots_corrected")
    parser.add_argument(
        "--correction",
        choices=["raw", "bh", "holm", "bonferroni"],
        default="bh",
        help="Which significance correction to display"
    )
    args = parser.parse_args()

    data = load_json(args.input)
    ensure_dir(args.output_dir)

    for experiment in sorted(data.keys()):
        rows = extract_rows_for_experiment(data, experiment)
        out_path = os.path.join(args.output_dir, f"{experiment}__forest_{args.correction}.png")
        forest_plot(
            rows=rows,
            title=f"Forest Plot: {experiment}",
            correction=args.correction,
            out_path=out_path,
        )
        print(f"Saved {out_path}")


if __name__ == "__main__":
    main()

How to read the plot

Each row is one cross-model comparison.

Example row:

chunking_gain | gpt_4o_mini vs route_llm

Interpretation:

point right of zero: first model has larger effect
point left of zero: second model has larger effect
CI crossing zero: uncertain difference
blue: significant under selected correction
gray: not significant
Recommended reporting use

This plot is ideal for a results section where you want to say:

which models differ significantly
in which cognitive-load effect
with what direction and uncertainty

It is much better than just dumping p-values in a table.
